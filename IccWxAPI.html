路径

wx.iisun.net/hpjr/index.php?C=?&M=?

基本返回信息

    error_code:  
        0：//执行成功，未发生任何错误
        -1://接受到的数据为空
        -2：//不属于任何团队
        -3://sql语句执行出错
        -4：//等级不足，没有操作权限
        -5://名称重复

        //1* 正则验证返回值
        -11 //电话格式出错
        -12//身份证格式出错

        -21://未查询到相关销售员的信息
        -22://已经评价过该销售

        //3* app 端
        -31 //提交至APP端 发生错误

        //4 
        -41 //已申请过该类型的贷款    
        -42 //审批已过,无法取消贷款
    status:接口状态 true/false (boolen)
    msg:返回信息 (ok,error sql ....)

            },
            '1':(管理员分配给客户经理)
            {
                'time':本次操作的时间,

            '2':(管理员分配给客户经理)
            {
                'time':本次操作的时间,
            },
            '3':(接收任务)
            {
                'time':本次操作的时间,
            },
            '4':(资料提醒)
            {
                'time':本次操作的时间,
                'msg':
                {
                    'need':(需要提醒的)['身份证','学生证',...],
                    'time':(预约时间),
                    'addr':(约见地址)
                }
            },
            '5':(审批反馈)
            {
                'time':本次操作的时间,
                'result':审批结果('pass'|'again'|'refuse'),
                'msg':审批结果的内容
            },
            '-1':(确认放款)
            {
                'time':本次操作的时间,
                'msg':
                {
                    'time':确认放款时间,
                    'money':放款金额 
                }
            },
            'mm': (占位，无意义),
        },
        {
            '1':(借款合同领取提醒)
            {
                'time':本次操作的时间,
                'msg':
                {
                    'time':领取时间,
                    'name':联系人,
                    'phone':联系电话,
                }
            },
            '2':(产权证提醒)
            {
                'time':本次操作的时间,
                'msg':
                {
                    'time':领取时间,
                    'name':联系人,
                    'phone':联系方式,
                }
            },
        }
    ]
}

    post:data
    {
        data:
        {
            'in_productid':开发商Id
            'in_buildid':楼盘内部id
            'money':贷款金额
            'premoney':首付金额
            'times':贷款期次
            'area':面积
            'bank':贷款银行
            'type':付款方式 （按揭0 /全款1）

            wx:来自哪个公众号 （zghg,zitan 。。。。）
        }

    {
        "status":true
    }
